class Solution:
    def containsDuplicate(self, nums):
        """
        :type nums: List[int]
        :rtype: bool
        """
        i = 0 
        nums = sorted(nums)
        flag = 0
        while i < len(nums) -1:
            if nums[i] == nums[i+1]:
                i += 1
                flag = 1
            else:
                i += 1
        if flag == 1:
            return 1 > 0
        else:
            return 1 < 0
