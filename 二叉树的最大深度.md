利用层次遍历，最后输出长度
```
# Definition for a binary tree node.
# class TreeNode(object):
#     def __init__(self, x):
#         self.val = x
#         self.left = None
#         self.right = None

class Solution(object):
    def maxDepth(self, root):
        """
        :type root: TreeNode
        :rtype: int
        """
        Quene = [root]
        result = []
        if not root :
            return len(result)
        while Quene:
            single_layer = []
            for i in range(0, len(Quene)):
                cur_node = Quene.pop(0)
                if cur_node.left:
                    Quene.append(cur_node.left)
                if cur_node.right:
                    Quene.append(cur_node.right)
                single_layer.append(cur_node.val)
            result.append(single_layer)
        return len(result)
```
