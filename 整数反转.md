基本思路就是转为字符串在转为整数
`
class Solution:
    def reverse(self, x: int) -> int:
        l = []
        s = 0
        t = abs(x)
        for i in range(len(str(t))):
            l.append(int(str(t)[i]))
            s += l[i] * 10 ** i
        if s > 2 ** 31 - 1 or s < - 2 ** 31:
            return 0
        if x > 0:
            return s
        else:
            return -s
        
`
