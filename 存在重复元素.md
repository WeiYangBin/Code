class Solution:
    def containsDuplicate(self, nums: List[int]) -> bool:
        # Method 2:
        dic = {}
        for i in range(len(nums)):
            if nums[i] in dic:
                dic[nums[i]] += 1
            else:
                dic[nums[i]] = 1
        print(dic)
        for key, value in enumerate(dic):
            if dic[value] != 1:
