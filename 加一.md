class Solution:
    def plusOne(self, digits: List[int]) -> List[int]:
        # 对于如果位数属于0-8之间，不需要进位的直接暴力+1
        digits[-1] = digits[-1] + 1
        if digits[-1] < 10:
            return digits
        # 对于9 + 1 = 10 => [1, 0] 直接利用先求和后拆分
        else:
            tmp = 0
            for i in range(len(digits)):
                tmp += digits[i] * 10**(len(digits)-i-1)
            res = []
            for i in range(len(str(tmp))):
                res.append(int(str(tmp)[i]))
            return res
